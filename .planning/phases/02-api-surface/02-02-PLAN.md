```
export const maxDuration = 60; // Vercel Pro for long Claude calls

export async function POST(request: NextRequest) {
  // 1. Rate limit check
  //    - Get IP from x-real-ip header, fall back to "127.0.0.1"
  //    - Call auditRateLimiter.limit(ip)
  //    - If !success, return 429 with Retry-After header (seconds until reset)

  // 2. Parse and validate request body
  //    - const body = await request.json()
  //    - Validate body.content exists and is a non-empty string
  //    - Return 422 VALIDATION_ERROR if invalid

  // 3. Check for existing audit (dedup via content hash)
  //    - Import buildCacheKey from @skillgate/audit-engine (re-export from hash.ts or use directly)
  //    - Build content hash: cacheKey = buildCacheKey(content), extract hash part (before the colon)
  //    - Check redis for hash-to-slug:{contentHash} -- if exists, this content was already audited
  //    - If slug found, load audit from slug:{slug} and audit:{contentHash}, return with cached: true

  // 4. Run audit
  //    - Call auditSkill(content) from @skillgate/audit-engine
  //    - Catch AuditError and map to appropriate error response using errorResponse()

  // 5. Generate slug and persist
  //    - buildSlug(content, contentHash)
  //    - Store to KV (all fire-and-forget, don't await):
  //      - slug:{slug} -> { contentHash, createdAt: new Date().toISOString() }
  //      - hash-to-slug:{contentHash} -> slug
  //      - audit:{contentHash} -> auditResult (redundant with engine cache, but ensures persistence)
  //    - Note: engine already caches via its own cache, but we also store under our key structure for slug lookups

  // 6. Return AuditResponse
  //    - Build meta: { slug, url: `/api/report/${slug}`, badge_url: `/api/badge/${slug}.svg`, created_at, cached: false }
  //    - Return Response.json({ result, meta })
}
```

Important: The content hash used for KV keys should be just the SHA-256 hex, NOT the full cache key with prompt version. Extract it: `buildCacheKey(content).split(":")[0]`.
