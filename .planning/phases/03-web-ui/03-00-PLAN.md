1. Install testing dependencies:
   ```
   pnpm add -D @testing-library/react @testing-library/jest-dom jsdom @vitejs/plugin-react --filter @skillgate/web
   ```
   Note: @vitejs/plugin-react is required for vitest to transform JSX in component tests.

2. Create `apps/web/vitest.config.ts`:
   ```typescript
   import { defineConfig } from "vitest/config";
   import react from "@vitejs/plugin-react";
   import path from "path";

   export default defineConfig({
     plugins: [react()],
     test: {
       environment: "jsdom",
       setupFiles: [],
       include: ["src/**/*.test.{ts,tsx}"],
       globals: true,
     },
     resolve: {
       alias: {
         "@": path.resolve(__dirname, "src"),
       },
     },
   });
   ```

   The `@` alias matches Next.js path resolution so test imports work identically to production code.
  </action>
